Code Implementation

function traverseRecursive(node, callback, depth = 0) {
    // Visit current node
    callback(node, depth, 'visit');
    
    // Process children recursively
    if (node.children && node.children.length > 0) {
        for (let i = 0; i < node.children.length; i++) {
            const child = node.children[i];
            callback(child, depth + 1, 'recurse');
            
            // Recursive call - function calls itself
            traverseRecursive(child, callback, depth + 1);
            
            callback(child, depth + 1, 'return');
        }
    }
    
    callback(node, depth, 'complete');
}

// Usage
traverseRecursive(rootNode, (node, depth, action) => {
    console.log(`${action}: ${node.name} at depth ${depth}`);
});
function traverseIterative(root, callback) {
    // Use explicit stack to simulate recursion
    const stack = [{ node: root, depth: 0, action: 'visit' }];
    
    while (stack.length > 0) {
        const { node, depth, action } = stack.pop();
        
        // Process current item
        callback(node, depth, action);
        
        // Add children to stack in reverse order
        // (to maintain left-to-right traversal)
        if (node.children && node.children.length > 0) {
            // Add completion marker first
            stack.push({ node, depth, action: 'complete' });
            
            // Add children in reverse order
            for (let i = node.children.length - 1; i >= 0; i--) {
                const child = node.children[i];
                stack.push({ node: child, depth: depth + 1, action: 'return' });
                stack.push({ node: child, depth: depth + 1, action: 'visit' });
            }
        }
    }
}

// Usage
traverseIterative(rootNode, (node, depth, action) => {
    console.log(`${action}: ${node.name} at depth ${depth}`);
});

Recursive Approach

  • Natural: Mirrors the tree structure naturally
  • Simple: Clean, readable code with automatic backtracking
  • Call Stack: Uses system call stack for state management
  • Risk: Stack overflow for very deep trees

Iterative Approach

  • Explicit Stack: Manual stack management for control
  • Memory Safe: No stack overflow risk
  • Performance: Better for deep trees and large datasets
  • Complexity: More complex code but predictable behavior
